% Command to simplify hyperlink syntax
%
% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% 

\define[2]\href{%
  \begingroup
    \setupinteraction[
      style=normal,
      color=TextColourHyperlink,
    ]%
    \goto{\color[TextColourHyperlink]{#1}}[url(#2)]%
  \endgroup%
}
